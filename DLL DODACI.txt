CREATE SEQUENCE SAHISTA_SEQ;

ALTER SEQUENCE SAHISTA_SEQ INCREMENT BY 4;
SELECT SAHISTA_SEQ.NEXTVAL FROM DUAL;
ALTER SEQUENCE SAHISTA_SEQ INCREMENT BY 1;

CREATE OR REPLACE TRIGGER SAHISTATrigger
BEFORE INSERT ON SAHISTA FOR EACH ROW
BEGIN
     SELECT SAHISTA_SEQ.NEXTVAL INTO :NEW.ID FROM DUAL;
END;
